    <script>
        // 网站域名
        const DOMAIN = "abc.woen.pics";
        
        // 获取DOM元素
        const passwordInput = document.getElementById('passwordInput');
        const submitBtn = document.getElementById('submitBtn');
        const resultMessage = document.getElementById('resultMessage');
        
        // 处理提交按钮点击
        submitBtn.addEventListener('click', checkPassword);
        
        // 处理回车键
        passwordInput.addEventListener('keypress', function(event) {
            if (event.key === 'Enter') {
                checkPassword();
            }
        });
        
        // 主函数：检查密码/文件名
        async function checkPassword() {
            // 获取输入值并清理
            const fileName = passwordInput.value.trim();
            
            // 验证输入
            if (!fileName) {
                showMessage("请输入文件名（不含.html扩展名）", false);
                passwordInput.focus();
                return;
            }
            
            // 禁用提交按钮并显示加载状态
            submitBtn.disabled = true;
            submitBtn.innerHTML = '<span class="loading"></span>检查中...';
            
            // 显示检查中的状态
            showMessage(`正在检查 https://${DOMAIN}/${fileName}.html ...`, true, "info");
            
            try {
                // 等待一小段时间，让用户看到检查中的状态
                await new Promise(resolve => setTimeout(resolve, 800));
                
                // 构建URL
                const targetUrl = `https://${DOMAIN}/${fileName}.html`;
                
                // 尝试检查文件是否存在
                const fileExists = await checkFileExists(targetUrl);
                
                if (fileExists) {
                    // 文件存在，跳转到该页面
                    showMessage(`验证成功！正在跳转到 ${fileName}.html ...`, true, "success");
                    
                    // 等待一会儿让用户看到成功消息
                    setTimeout(() => {
                        window.location.href = targetUrl;
                    }, 1500);
                } else {
                    // 文件不存在，显示错误
                    showMessage(`密码错误：未找到文件 ${fileName}.html`, false);
                    
                    // 重新启用提交按钮
                    submitBtn.disabled = false;
                    submitBtn.innerHTML = '确定';
                    
                    // 清空输入框并聚焦
                    passwordInput.value = '';
                    passwordInput.focus();
                }
            } catch (error) {
                // 发生错误
                showMessage(`检查文件时发生错误：${error.message}`, false);
                
                // 重新启用提交按钮
                submitBtn.disabled = false;
                submitBtn.innerHTML = '确定';
            }
        }
        
        // 检查文件是否存在
        async function checkFileExists(url) {
            try {
                // 使用HEAD请求检查文件是否存在（不下载整个文件）
                const response = await fetch(url, { 
                    method: 'HEAD',
                    mode: 'cors', // 尝试CORS请求
                    cache: 'no-cache'
                });
                
                // 如果状态码是200，则文件存在
                return response.ok;
            } catch (error) {
                // 如果请求失败，可能是网络问题或CORS限制
                // 尝试使用GET请求作为备用方案
                try {
                    const response = await fetch(url, { 
                        method: 'GET',
                        mode: 'no-cors', // 不检查CORS
                        cache: 'no-cache'
                    });
                    
                    // 对于no-cors模式，我们无法读取响应状态
                    // 但我们可以检查响应类型
                    return response.type !== 'error';
                } catch (secondError) {
                    // 如果两种方法都失败，则假设文件不存在
                    return false;
                }
            }
        }
        
        // 显示消息
        function showMessage(message, isSuccess, type = "error") {
            resultMessage.textContent = message;
            resultMessage.className = "result-message";
            
            if (type === "success") {
                resultMessage.classList.add("success");
            } else if (type === "info") {
                // 信息状态（检查中）
                resultMessage.style.backgroundColor = "rgba(33, 150, 243, 0.1)";
                resultMessage.style.color = "#0d47a1";
                resultMessage.style.border = "1px solid #2196f3";
            } else {
                resultMessage.classList.add("error");
            }
            
            resultMessage.style.display = "block";
            
            // 如果不是检查中的状态，5秒后自动隐藏消息
            if (type !== "info") {
                setTimeout(() => {
                    resultMessage.style.display = "none";
                }, 5000);
            }
        }
        
        // 页面加载完成后聚焦到输入框
        window.addEventListener('DOMContentLoaded', () => {
            passwordInput.focus();
        });
    </script>
